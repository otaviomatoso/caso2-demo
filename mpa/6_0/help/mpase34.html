Figura A.2: Exemplo de script disponível na IDE
 
scripts.add{ 
  name   = ”Nome do Script”, 
  author = ”Nome do Autor do Script (e-mail do autor)”, 
 
  disable = true,  -- desabilita o script, nao o deixa disponivel para ser 
                   -- utilizado pelo usuario no seletor de scripts do 
                   -- Cliente MPA 
 
  description = [[ 
    Descricao e outras informacoes sobre funcionamento do Script 
  ]], 
 
  callback = function(self) -- funcao que vai ser chamada pelo seletor de 
                            -- script do Cliente MPA, o parametro self e 
                            -- opcional, mas permite que sejam desenvolvidas 
                            -- funcoes auxiliares 
 
    MPA.PrintOut( ”:: Script de exemplo ::\n” ) 
    MPA.PrintOut( ”Nome do Script: ”..self.name ) 
    MPA.PrintOut( ”Criado por: ”..self.author ) 
 
    local classid = ”BOOLEAN_POINT” 
 
    local n = 3 
 
    for i = 1, n do 
      MPA.AddConfig( classid, ”Point_”..i ) 
    end 
 
    local ClassPoints = MPA.GetConfigs( classid ) 
 
    if #ClassPoints ˜= n then 
      return nil 
    end 
 
    if not MPA.IsConfigNameUsed( classid, ”Point_”..n ) then 
      return nil 
    end 
 
    MPA.ResetConfigs( classid ) 
 
    if MPA.IsConfigNameUsed( classid, ”Point_”..n ) then 
      return nil 
    end 
 
    return true 
  end 
}
